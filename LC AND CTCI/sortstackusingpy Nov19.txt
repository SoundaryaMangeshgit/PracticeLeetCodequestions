# CTCI sort a stack using 
stack1=[5,6,3,2,9,1]
new_len=len(stack1)
temp=[]
min=stack1[0]
while(len(temp)!=new_len):
    
    min=stack1[0]
    
    for i in range(1,len(stack1)):
        if(min >stack1[i]):
            min=stack1[i]
    temp.append(min)
    stack1.remove(min)
   
    
print("sorted stack",temp)
