using System; 
using System.Collections.Generic; 
//sort stack such that find minimum nd then push into stack 
class StackSort 
{ 

 public static Stack<int> ToSortStack(Stack<int> input_stack)
     {
   
   		Stack<int> temp_Stack=new Stack<int>();
        int temp;
       while(input_stack.Count>0)
       {  
          temp=input_stack.Pop();
       
   
       while(temp_Stack.Count>0 && temp_Stack.Peek()>temp)
       {
        
         input_stack.Push(temp_Stack.Pop());
        
       }
        
       temp_Stack.Push(temp);
       }      
   return temp_Stack;
   
     }
 
  public static void Main() 
  { 
	Stack<int> input_stack=new Stack<int>();
    input_stack.Push(5);
    input_stack.Push(6);
    input_stack.Push(3);
    input_stack.Push(2);
    input_stack.Push(9);
    input_stack.Push(1);
    Stack<int> SortedStack=ToSortStack(input_stack);
    while(SortedStack.Count>0)
      {
		Console.WriteLine(SortedStack.Pop());
      }

  } 
}
